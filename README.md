```
terraform {                           # Начало блока настройки Terraform: задаём глобальные параметры конфигурации
  required_providers {                # Блок перечисления провайдеров, которые требуются для работы конфигурации
    yandex = {                        # Объявление провайдера Yandex Cloud под именем "yandex"
      source  = "yandex-cloud/yandex" # Указываем источник провайдера — официальный репозиторий Yandex Cloud для Terraform
      version = ">= 0.13"             # Указываем версию провайдера: ~> 0.114 означает «версия 0.114 или выше, но без смены мажорной версии»
    }
  }
}

provider "yandex" {            # Начало блока настройки провайдера Yandex Cloud
  token     = var.YC_TOKEN     # Передаём OAuth‑токен или API‑ключ для доступа к Yandex Cloud (берётся из переменных)
  cloud_id  = var.YC_CLOUD_ID  # Передаём идентификатор облака в Yandex Cloud (из переменных)
  folder_id = var.YC_FOLDER_ID # Передаём идентификатор каталога в облаке (из переменных)
  zone      = var.yc_zone      # Указываем зону доступности, где будут создаваться ресурсы (из переменных, по умолчанию ru‑central1‑a)
}



resource "yandex_compute_instance" "vm" { # Объявляем ресурс «виртуальная машина» в Yandex Cloud с локальным именем "vm" в Terraform
  count = 2
  name = "vm-${count.index}"           # Задаём имя виртуальной машины в Yandex Cloud: "ubuntu-docker-zabbix"

  resources {  # Начинаем блок описания вычислительных ресурсов виртуальной машины
    cores  = 2 # Указываем количество виртуальных ядер CPU: 2 ядра (минимальная конфигурация)
    memory = 2 # Указываем объём оперативной памяти: 2 ГБ (минимальная конфигурация)
  }

  boot_disk {                           # Начинаем блок настройки загрузочного диска виртуальной машины
    initialize_params {                 # Начинаем блок параметров инициализации диска при создании
      image_id = "fd8jqd7hb16epiac8lla" # Указываем ID образа ОС: Ubuntu 24.04 LTS (нужно уточнить актуальный в консоли Yandex Cloud)
      size     = 20                     # Указываем размер диска в ГБ: 20 ГБ
    }
  }

  network_interface {                         # Начинаем блок настройки сетевого интерфейса виртуальной машины
    subnet_id = var.subnet_id # Указываем ID подсети, к которой подключается ВМ (ссылка на созданный ресурс подсети)
    nat       = true                          # Включаем NAT: позволяет ВМ иметь доступ в интернет через публичный IP
  }

  metadata = { # Начинаем блок метаданных — дополнительных параметров, передаваемых в ОС при запуске
    ssh-keys = "ubuntu:${file(var.ssh_public_key)}"  # Задаём SSH‑ключ для доступа к ВМ: пользователь ubuntu и ключ из переменной
    #user-data = file("${path.module}/cloud-init.yaml") # Указываем путь к скрипту настройки ОС, который выполнится при первом запуске ВМ
  }

  scheduling_policy {
    preemptible = true
  }

}


resource "yandex_lb_target_group" "tg" {
  count = 1                                                           # одна таргет-группа
  name  = "lb-target-group"                                          # имя таргет-группы

  dynamic "target" {
    for_each = yandex_compute_instance.vm                            # добавляем все созданные ВМ в группу
    content {
      subnet_id = var.subnet_id      # ID подсети целевой ВМ
      address   = target.value.network_interface[0].ip_address       # IP-адрес целевой ВМ
    }
  }
}

resource "yandex_lb_network_load_balancer" "nlb" {
  name = "http-nlb"                                                  # имя балансировщика нагрузки

  listener {
    name        = "http-listener"                                    # имя слушателя
    port        = 80                                                  # порт приема трафика
    target_port = 80                                                  # порт отправки на ВМ
    protocol    = "tcp"                                               # протокол TCP
  }

  attached_target_group {
    target_group_id = yandex_lb_target_group.tg[0].id                # привязываем таргет-группу

    healthcheck {
      name            = "http-healthcheck"                                    # имя проверки работоспособности
      http_options {
        port          = 80                                                    # порт для healthcheck
        path          = "/"                                                   # путь для запроса
      }
    }
  }
}
```
